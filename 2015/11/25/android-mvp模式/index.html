android mvp模式

android mvp模式简介

MVP模式是MVC模式在Android上的一种变体。在MVC模式中，Activity应该是属于View这一层的，而实质上，它同时还承担了Controller的功能，耦合度太高。把Activity的View和Controller抽离出来形成View和Presenter，降低耦合性。
MVP是Activity中的UI逻辑抽象成VIew接口，把业务逻辑抽象成Presenter接口，Activity只用来响应生命周期，其它工作丢到Presenter中去完成。Presenter是Model和View之间的桥梁，View不直接操作Model。

MVP模式简单实例

Activity:LoginActivity只做了findView以及setListener的工作，而且包含了一个ILoginPresenter，所有业务逻辑都是通过调用ILoginPresenter的具体接口来完成。

//View接口
public interface ILoginView {
public boolean checkInputInfo();
public void clearInputInfo();
public void onLogResult(boolean result,int code);
}    

public class LoginActivity extends Activity implements ILoginView,View.OnClickListener{

private ILogPresenter iLogPresenter;
private EditText et_username;
private EditText et_userpwd;
private Button bn_log;
private LogUser user;
@Override
protected void onCreate(Bundle savedInstanceState) {
    super.onCreate(savedInstanceState);
    setContentView(R.layout.activity_login);
    init();
}
//初始化控件
public void init(){
    et_username = (EditText)findViewById(R.id.et_LoginUsername);
    et_userpwd = (EditText)findViewById(R.id.et_Pwd);
    bn_log = (Button)findViewById(R.id.bn_Login);
    iLogPresenter = new LogPresenter(this);
    bn_log.setOnClickListener(this);
    user = new LogUser();
}

@Override
public boolean checkInputInfo() {
  if(!TextUtils.isEmpty(et_username.getText().toString())&&!TextUtils.isEmpty(et_userpwd.getText().toString())){
      return true;
  }else if(TextUtils.isEmpty(et_username.getText().toString())){
      Toast.makeText(LoginActivity.this,"请输入用户名",Toast.LENGTH_LONG).show();
      return false;
  }else if(TextUtils.isEmpty(et_userpwd.getText().toString())){
      Toast.makeText(LoginActivity.this,"请输入密码",Toast.LENGTH_LONG).show();
      return false;
  }
    return false;
}

@Override
public void clearInputInfo() {
    et_username.setText("");
    et_userpwd.setText("");
}

@Override
public void onLogResult(boolean result, int code) {
    if(result){
        Toast.makeText(LoginActivity.this,"登陆成功",Toast.LENGTH_LONG).show();
        Intent itent = new Intent(LoginActivity.this,MainActivity.class);
        startActivity(itent);
        finish();
    }else{
        Toast.makeText(LoginActivity.this,"登陆失败",Toast.LENGTH_LONG).show();
    }
}

@Override
public void onClick(View v) {
    switch(v.getId()){
        case R.id.bn_Login:
            user.setUsername(et_username.getText().toString());
            user.setUserpwd(et_userpwd.getText().toString());
            iLogPresenter.doLogin(user);
            break;
    }
}
}

Presenter：LoginPresenter保留了ILoginView的引用，因此在LoginPresenter里就可以直接进行UI操作了，而不用在Activity里完成。这里使用了ILoginView引用，而不是直接使用Activity，这样一来，如果在别的Activity里也需要用到相同的业务逻辑，就可以直接复用LoginPresenter类了

//Presenter接口
public interface ILogPresenter {
   public boolean checkInputInfo();
   public void clearInputInfo();
   public void doLogin(LogUser user);
}

public class LogPresenter implements ILogPresenter{
private ILoginView iLoginView;

public LogPresenter(ILoginView iLoginView){
   this.iLoginView = iLoginView;
}

@Override
public boolean checkInputInfo() {
    return iLoginView.checkInputInfo();
}

@Override
public void clearInputInfo() {
    iLoginView.clearInputInfo();
}

@Override
public void doLogin(LogUser user) {
    if(!checkInputInfo())return;
    if(user.getUsername().equals("admin")&&user.getUserpwd().equals("admin")){
        iLoginView.onLogResult(true,0);
    }else{
        iLoginView.onLogResult(false,1);
    }
}
}
